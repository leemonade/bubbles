## Overview

ContextContainer is a component to stack and divide elements inside, with padding and dividers.

<ThemeProvider>
  <ContextContainer
    title="Context title"
    subtitle="Context subtitle"
    description="Context description"
    divided
  >
    <TextInput headerStyle={{ width: 300 }} label="Label" />
    <TextInput headerStyle={{ width: 300 }} label="Label" />
  </ContextContainer>
</ThemeProvider>

## ContextContainer direction

ContextContainer has 4 different directions: `column`, `row`, `row-reverse`, `column-reverse`

<ThemeProvider>
  <Box style={{ display: 'flex', flexDirection: 'column', gap: 32 }}>
    <ContextContainer title="Context title">
      <TextInput headerStyle={{ width: 300 }} label="Label 1" />
      <TextInput headerStyle={{ width: 300 }} label="Label 2" />
      <TextInput headerStyle={{ width: 300 }} label="Label 3" />
    </ContextContainer>
    <ContextContainer title="Context title" direction="column-reverse">
      <TextInput headerStyle={{ width: 300 }} label="Label 1" />
      <TextInput headerStyle={{ width: 300 }} label="Label 2" />
      <TextInput headerStyle={{ width: 300 }} label="Label 3" />
    </ContextContainer>
    <ContextContainer title="Context title" direction="row">
      <TextInput headerStyle={{ width: 300 }} label="Label 1 " />
      <TextInput headerStyle={{ width: 300 }} label="Label 2" />
      <TextInput headerStyle={{ width: 300 }} label="Label 3" />
    </ContextContainer>
    <ContextContainer title="Context title" direction="row-reverse">
      <TextInput headerStyle={{ width: 300 }} label="Label 1" />
      <TextInput headerStyle={{ width: 300 }} label="Label 2" />
      <TextInput headerStyle={{ width: 300 }} label="Label 3" />
    </ContextContainer>
  </Box>
